		//Expected: 20

2. Second Largest Element in an Array:

		//Given:
		int[] a2 = { 15, 5, 20, 12, 14, 20, 8 };
		// Expected: 10

3. Check if array is sorted or not (Ascending)

		// Given:
		int[] a3 = { 1,2,3,4,5,6,7,8,9 };
		// Expected: true
		//Given:
		int[] a4 = { 1,8,3,4,5,6,7,8,9 };
		// Expected: false

		// Given:
